		main() 
		{ 
		int n,i,x[10],max,min; 
		clrscr(); 
		printf("\nEnter how many Numbers:"); 
		scanf("%d",&n); 
		printf("\nEnter %d Numbers:",n); 
		for(i=1;i<=n;i++) 
		scanf("%d",&x[i]); 
		max=x[1]; min=x[1];
		for(i=1;i<=n;i++) 
		{ 
		if(x[i]>max) max=x[i]; 
		if(x[i]<min) min=x[i]; 
		} 
		printf("\nLARGEST NUMBER:%d",max); 
		printf("\nSMALLEST NUMBER:%d",min); 
		}
